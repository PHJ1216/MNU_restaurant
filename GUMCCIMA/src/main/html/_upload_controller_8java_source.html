108 String uploadFolderPath = getFolder();
109 // make folder --------
110 File uploadPath = new File(uploadFolder, uploadFolderPath);
111
112 if (uploadPath.exists() == false) {
113 uploadPath.mkdirs();

115 // make yyyy/MM/dd folder
116
117 for (MultipartFile multipartFile : uploadFile) {

119 AttachFileDTO attachDTO = new AttachFileDTO();

121 String uploadFileName = multipartFile.getOriginalFilename();
122
123 // IE has file path
124 uploadFileName = uploadFileName.substring(uploadFileName.lastIndexOf("\\") + 1);
125 log.info("only file name: " + uploadFileName);
126 attachDTO.setFileName(uploadFileName);
127
128 UUID uuid = UUID.randomUUID();
129
130 uploadFileName = uuid.toString() + "_" + uploadFileName;
131
132 try {
133 File saveFile = new File(uploadPath, uploadFileName);
134 multipartFile.transferTo(saveFile);
135
136 attachDTO.setUuid(uuid.toString());
137 attachDTO.setUploadPath(uploadFolderPath);
138
139 // check image type file
140 if (checkImageType(saveFile)) {

142 attachDTO.setImage(true);

144 FileOutputStream thumbnail = new FileOutputStream(new File(uploadPath, "s_" + uploadFileName));

146 Thumbnailator.createThumbnail(multipartFile.getInputStream(), thumbnail, 100, 100);

186 @GetMapping(value = "/download", produces = MediaType.APPLICATION_OCTET_STREAM_VALUE)
187 @ResponseBody
188 public ResponseEntity<Resource> downloadFile(@RequestHeader("User-Agent") String userAgent, String fileName) {
189
190 Resource resource = new FileSystemResource("c:\\upload\\" + fileName);
191
192 if (resource.exists() == false) {
193 return new ResponseEntity<>(HttpStatus.NOT_FOUND);
194 }
195
196 String resourceName = resource.getFilename();
197
198 // remove UUID
199 String resourceOriginalName = resourceName.substring(resourceName.indexOf("_") + 1);
200
201 HttpHeaders headers = new HttpHeaders();
202 try {
203
204 boolean checkIE = (userAgent.indexOf("MSIE") > -1 || userAgent.indexOf("Trident") > -1);
205
206 String downloadName = null;
207
208 if (checkIE) {
209 downloadName = URLEncoder.encode(resourceOriginalName, "UTF8").replaceAll("\\+", " ");
210 } else {
211 downloadName = new String(resourceOriginalName.getBytes("UTF-8"), "ISO-8859-1");
212 }
213
214 headers.add("Content-Disposition", "attachment; filename=" + downloadName);
215
216 } catch (UnsupportedEncodingException e) {
217 e.printStackTrace();
218 }
219
220 return new ResponseEntity<Resource>(resource, headers, HttpStatus.OK);
221 }
